// Sistema de gestión académica
class CampusManager {
  function loadDocentes($cursoId) {
    $db = Database::connect();
    $docentes = $db->query("SELECT * FROM docentes WHERE curso_id = ?", [$cursoId]);
    return $docentes;
  }
  function generarPlantilla($datos) {
    // Exportación PDF con jsPDF
    $pdf = new jsPDF($datos);
    return $pdf->export("plantilla_generada.pdf");
  }
}
